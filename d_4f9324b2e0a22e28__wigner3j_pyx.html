9cdef double _coeffc(long j1, long j2, long j3, long m3) nogil: 

10 r""" 

11 Coefficient C for the recursion relation of the Wigner 3j-symbols 

12  

13 See equation (7) in [#]_. 

14  

15 References: 

16 .. [#] `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_ 

17 """ 

18 cdef double x = ( 

19 (j3 * j3 - (j1 - j2) * (j1 - j2)) 

20 * <double>((j1 + j2 + 1) * (j1 + j2 + 1) - j3 * j3) 

21 * (j3 * j3 - m3 * m3) 

22 ) 

23 return sqrt(x)

26cdef long _coeffd(long j1, long j2, long j3, long m1, long m2, long m3) nogil: 

27 r""" 

28 Coefficient D for the recursion relation of the Wigner 3j-symbols 

29  

30 See equation (7) in [#]_. 

31  

32 References: 

33 .. [#] `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_ 

34 """ 

35 return (2 * j3 + 1) * ( 

36 j3 * (j3 + 1) * (m2 - m1) + j2 * (j2 + 1) * m3 - j1 * (j1 + 1) * m3 

37 )

40cdef double _initforwardm(long j1, long j2, long m1, long m2) nogil: 

41 """ 

42 Initial value for the forward recursion for j3 = |m1 + m2| 

43  

44 See equation (16) in [#]_. 

45  

46 References: 

47 .. [#] `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_ 

48 """ 

49 return minusonepow(j2 + m2) * exp( 

50 ( 

51 lgamma(j1 + m1 + 1) 

52 + lgamma(j2 + m2 + 1) 

53 + lgamma(j1 + j2 - m1 - m2 + 1) 

54 + lgamma(2 * m1 + 2 * m2 + 1) 

55 - lgamma(j1 - m1 + 1) 

56 - lgamma(j2 - m2 + 1) 

57 - lgamma(j1 - j2 + m1 + m2 + 1) 

58 - lgamma(j2 - j1 + m1 + m2 + 1) 

59 - lgamma(j1 + j2 + m1 + m2 + 2) 

60 ) 

61 * 0.5 

62 )

66 """ 

67 Initial value for the forward recursion, if j3 = |j1 - j2|

69 See equation (14) in [#]_. 

70  

71 References: 

72 .. [#] `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_

79 + lgamma(2 * j2 + 1)

90cdef double _wigner3jforward( 

91 long j1, long j2, long j3, 

92 long m1, long m2, long m3, 

93 double *cache 

94) nogil: 

95 """ 

96 Forward recursion for Wigner 3j symbol 

97  

98 See equation (6) in [#]_. The cache contains already computed values or nan, with 

99 j3 as index. It must be j3 + 1 elements large. 

100  

101 References: 

102 .. [#] `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_ 

103 """ 

104 if not isnan(cache[j3]): 

105 return cache[j3] 

106 cdef long j3min = max(labs(j1 - j2), labs(m1 + m2)) 

107 cdef double res 

108 if j3 == j3min: # TODO: Unnecessary for possible values of j3? 

109 if j3 == j1 - j2: 

110 res = _initforwardj(j1, j2, m1, m2) 

111 elif j3 == j2 - j1: 

112 res = _initforwardj(j2, j1, m2, m1) 

113 elif j3 == m1 + m2: 

114 res = _initforwardm(j1, j2, m1, m2) 

115 else: # j3 == -m1 - m2: 

116 res = _initforwardm(j2, j1, -m2, -m1) 

117 cache[j3] = res 

118 return res 

119 # This if avoids divide by zero in the following if 

120 if j3min == 0 and j3 == 1: 

121 res = -_wigner3jforward(j1, j2, 0, m1, m2, m3, cache) * (m2 - m1) / _coeffc(j1, j2, 1, m3) 

122 cache[j3] = res 

123 return res 

124 cdef double res_prev = -_coeffd(j1, j2, j3 - 1, m1, m2, m3) * _wigner3jforward( 

125 j1, j2, j3 - 1, m1, m2, m3, cache 

126 ) 

127 if j3 == j3min + 1: 

128 res = res_prev / (j3min * _coeffc(j1, j2, j3, m3)) 

129 cache[j3] = res 

130 return res 

131 cdef double res_prev_prev = ( 

132 -j3 

133 * _coeffc(j1, j2, j3 - 1, m3) 

134 * _wigner3jforward(j1, j2, j3 - 2, m1, m2, m3, cache) 

135 ) 

136 res = (res_prev + res_prev_prev) / ((j3 - 1) * _coeffc(j1, j2, j3, m3)) 

137 cache[j3] = res 

138 return res

141cdef double _wigner3jbackward( 

142 long j1, long j2, long j3, 

143 long m1, long m2, long m3, 

144 double *cache 

145) nogil: 

146 """ 

147 Backward recursion for Wigner 3j symbol 

148  

149 See equation (6) in [#]_. The cache contains already computed values or nan, with 

150 j3 as index. It must be j1 + j2 + 1 elements large. 

151  

152 References: 

153 .. [#] `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_ 

154 """ 

155 if not isnan(cache[j3]): 

156 return cache[j3] 

157 cdef double res, res_prev 

158 if j3 == j1 + j2: 

159 res = minusonepow(j3 - m3) * exp( 

160 ( 

161 lgamma(2 * j1 + 1) 

162 + lgamma(2 * j2 + 1) 

163 + lgamma(j3 + m3 + 1) 

164 + lgamma(j3 - m3 + 1) 

165 - lgamma(2 * j3 + 2) 

166 - lgamma(j1 - m1 + 1) 

167 - lgamma(j1 + m1 + 1) 

168 - lgamma(j2 - m2 + 1) 

169 - lgamma(j2 + m2 + 1) 

170 ) 

171 * 0.5 

172 ) 

173 cache[j3] = res 

174 return res 

175 res_prev = -_coeffd(j1, j2, j3 + 1, m1, m2, m3) * _wigner3jbackward( 

176 j1, j2, j3 + 1, m1, m2, m3, cache 

177 ) 

178 if j3 == j1 + j2 - 1: 

179 res = res_prev / ((j3 + 2) * _coeffc(j1, j2, j3 + 1, m3)) 

180 cache[j3] = res 

181 return res 

182 cdef double res_prev_prev = ( 

183 -(j3 + 1) 

184 * _coeffc(j1, j2, j3 + 2, m3) 

185 * _wigner3jbackward(j1, j2, j3 + 2, m1, m2, m3, cache) 

186 ) 

187 res = (res_prev + res_prev_prev) / ((j3 + 2) * _coeffc(j1, j2, j3 + 1, m3)) 

188 cache[j3] = res 

189 return res

192cdef double wigner3j(long j1, long j2, long j3, long m1, long m2, long m3) nogil: 

193 r""" 

194 Wigner-3j symbol 

195  

196 Calculate 

197  

198 .. math:: 

199  

200 \begin{pmatrix} j_1 & j_2 & j_3 \\ m_1 & m_2 & m_3 \end{pmatrix} 

201  

202 recursively by forward or backward recurstion. 

203 Starting points are the extremal values for `j3.` The recursive function 

204 calls are cached. For unphysical value combinations `0.0` is returned, 

205 similar to Mathematica's behavior. 

206  

207 References: 

208 - `Y.-L. Xu, J. Comput. Phys. 139, 137 - 165 (1998) <https://doi.org/10.1006/jcph.1997.5867>`_ 

209 - `DLMF 34.3 <https://dlmf.nist.gov/34.3>`_ 

210 """ 

211 if j1 < 0 or j2 < 0 or j3 < 0: 

212 raise ValueError("j values must be non-negative") 

213 if not labs(j1 - j2) <= j3 <= j1 + j2: 

214 return 0.0 

215 if labs(m1) > j1 or labs(m2) > j2 or labs(m3) > j3: 

216 return 0.0 

217 if m1 + m2 + m3 != 0: 

218 return 0.0 

219  

220 cdef double *cache = <double*>malloc((j1 + j2 + 1) * sizeof(double)) # Size could be smaller by closely inspecting forward or backward recursion 

221 cdef long i 

222 for i in range(j1 + j2 + 1): 

223 cache[i] = <double>NAN 

224  

225 cdef double res 

226 if (j1 + j2 - labs(j1 - j2)) // 4 + labs(j1 - j2) > j3: 

227 res = _wigner3jforward(j1, j2, j3, m1, m2, m3, cache) 

228 else: 

229 res = _wigner3jbackward(j1, j2, j3, m1, m2, m3, cache) 

230 free(cache) 

231 return res
